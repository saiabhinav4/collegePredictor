    <!-- Create a script tag and write the JavaScript code to generate the PDF -->
    <script>
        const downloadBtn = document.getElementById("download-btn");
        const table = document.querySelector("table");

        downloadBtn.addEventListener("click", () => {
            const doc = new jsPDF();
            const data = [];

            // Add header row
            const header = [];
            for (let i = 0; i < table.rows[0].cells.length; i++) {
                header.push(table.rows[0].cells[i].textContent);
            }
            data.push(header);

            // Add data rows
            for (let i = 1; i < table.rows.length; i++) {
                const tableRow = table.rows[i];
                const rowData = [];
                for (let j = 0; j < tableRow.cells.length; j++) {
                    rowData.push(tableRow.cells[j].textContent);
                }
                data.push(rowData);
            }

            // Set column widths
            const columnWidths = [];
            for (let i = 0; i < table.rows[0].cells.length; i++) {
                const cellWidth = table.rows[0].cells[i].clientWidth / 10;
                columnWidths.push(cellWidth);
            }

            // Create table
            doc.autoTable({
                head: data.slice(0, 1),
                body: data.slice(1),
                startY: 20,
                headStyles: {
                    fillColor: [100, 100, 255],
                    textColor: 255,
                    fontSize: 12,
                    fontStyle: "bold",
                    halign: "center",
                },
                bodyStyles: {
                    fontSize: 10,
                    halign: "center",
                },
                columnStyles: columnWidths,
            });

            // Download the PDF
            doc.save("table.pdf");
        });
    </script>
